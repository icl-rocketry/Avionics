469 inline uint32_t sdCardCapacity(csd_t* csd) {
470  if (csd->v1.csd_ver == 0) {
471  uint8_t read_bl_len = csd->v1.read_bl_len;
472  uint16_t c_size = (csd->v1.c_size_high << 10)
473  | (csd->v1.c_size_mid << 2) | csd->v1.c_size_low;
474  uint8_t c_size_mult = (csd->v1.c_size_mult_high << 1)
475  | csd->v1.c_size_mult_low;
476  return (uint32_t)(c_size + 1) << (c_size_mult + read_bl_len - 7);
477  } else if (csd->v2.csd_ver == 1) {
478  uint32_t c_size = 0X10000L * csd->v2.c_size_high + 0X100L
479  * (uint32_t)csd->v2.c_size_mid + csd->v2.c_size_low;
480  return (c_size + 1) << 10;
481  } else {
482  return 0;
483  }
