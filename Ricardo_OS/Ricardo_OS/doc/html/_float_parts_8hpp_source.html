14 struct FloatParts {

20  FloatParts(TFloat value) {
21  uint32_t maxDecimalPart = sizeof(TFloat) >= 8 ? 1000000000 : 1000000;
22  decimalPlaces = sizeof(TFloat) >= 8 ? 9 : 6;
23 
24  exponent = normalize(value);
25 
26  integral = uint32_t(value);
27  // reduce number of decimal places by the number of integral places
28  for (uint32_t tmp = integral; tmp >= 10; tmp /= 10) {
29  maxDecimalPart /= 10;
30  decimalPlaces--;
31  }
32 
33  TFloat remainder = (value - TFloat(integral)) * TFloat(maxDecimalPart);
34 
35  decimal = uint32_t(remainder);
36  remainder = remainder - TFloat(decimal);
37 
38  // rounding:
39  // increment by 1 if remainder >= 0.5
40  decimal += uint32_t(remainder * 2);
41  if (decimal >= maxDecimalPart) {
42  decimal = 0;
43  integral++;
44  if (exponent && integral >= 10) {
45  exponent++;
46  integral = 1;
47  }
48  }
49 
50  // remove trailing zeros
51  while (decimal % 10 == 0 && decimalPlaces > 0) {
52  decimal /= 10;
53  decimalPlaces--;
54  }
55  }

57  static int16_t normalize(TFloat& value) {
58  typedef FloatTraits<TFloat> traits;
59  int16_t powersOf10 = 0;
60 
61  int8_t index = sizeof(TFloat) == 8 ? 8 : 5;
62  int bit = 1 << index;
63 
64  if (value >= ARDUINOJSON_POSITIVE_EXPONENTIATION_THRESHOLD) {
65  for (; index >= 0; index--) {
66  if (value >= traits::positiveBinaryPowerOfTen(index)) {
67  value *= traits::negativeBinaryPowerOfTen(index);
68  powersOf10 = int16_t(powersOf10 + bit);
69  }
70  bit >>= 1;
71  }
72  }
73 
74  if (value > 0 && value <= ARDUINOJSON_NEGATIVE_EXPONENTIATION_THRESHOLD) {
75  for (; index >= 0; index--) {
76  if (value < traits::negativeBinaryPowerOfTenPlusOne(index)) {
77  value *= traits::positiveBinaryPowerOfTen(index);
78  powersOf10 = int16_t(powersOf10 - bit);
79  }
80  bit >>= 1;
81  }
82  }
83 
84  return powersOf10;
85  }
