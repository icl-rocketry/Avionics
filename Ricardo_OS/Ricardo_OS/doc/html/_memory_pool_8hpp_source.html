122  // Squash the free space between strings and variants
123  //
124  // _begin _end
125  // v v
126  // +-------------+--------------+
127  // | strings... | ...variants |
128  // +-------------+--------------+
129  // ^
130  // _left _right
131  //
132  // This funcion is called before a realloc.
133  ptrdiff_t squash() {
134  char* new_right = addPadding(_left);
135  if (new_right >= _right)
136  return 0;
137 
138  size_t right_size = static_cast<size_t>(_end - _right);
139  memmove(new_right, _right, right_size);
140 
141  ptrdiff_t bytes_reclaimed = _right - new_right;
142  _right = new_right;
143  _end = new_right + right_size;
144  return bytes_reclaimed;
145  }

165  template <typename TIterator>
166  const char* findString(TIterator str) {
167  for (char* next = _begin; next < _left; ++next) {
168  char* begin = next;

170  // try to match

172  if (*next++ == 0)
