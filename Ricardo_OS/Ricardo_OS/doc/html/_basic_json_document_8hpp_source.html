11 // Helper to implement the "base-from-member" idiom

106  void shrinkToFit() {
107  ptrdiff_t bytes_reclaimed = _pool.squash();
108  if (bytes_reclaimed == 0)
109  return;
110 
111  void* old_ptr = _pool.buffer();
112  void* new_ptr = this->reallocate(old_ptr, _pool.capacity());
113 
114  ptrdiff_t ptr_offset =
115  static_cast<char*>(new_ptr) - static_cast<char*>(old_ptr);
116 
117  _pool.movePointers(ptr_offset);
118  _data.movePointers(ptr_offset, ptr_offset - bytes_reclaimed);
119  }
