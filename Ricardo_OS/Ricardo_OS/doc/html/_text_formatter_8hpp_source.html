53  template <typename T>
54  void writeFloat(T value) {
55  if (isnan(value))
56  return writeRaw(ARDUINOJSON_ENABLE_NAN ? "NaN" : "null");
57 
58 #if ARDUINOJSON_ENABLE_INFINITY
59  if (value < 0.0) {
60  writeRaw('-');
61  value = -value;
62  }
63 
64  if (isinf(value))
65  return writeRaw("Infinity");
66 #else
67  if (isinf(value))
68  return writeRaw("null");
69 
70  if (value < 0.0) {
71  writeRaw('-');
72  value = -value;
73  }
74 #endif
75 
76  FloatParts<T> parts(value);
77 
78  writePositiveInteger(parts.integral);
79  if (parts.decimalPlaces)
80  writeDecimals(parts.decimal, parts.decimalPlaces);

84  writePositiveInteger(-parts.exponent);

98  template <typename T>
99  void writePositiveInteger(T value) {
100  char buffer[22];
101  char *end = buffer + sizeof(buffer);
102  char *begin = end;
103 
104  // write the string in reverse order
105  do {
106  *--begin = char(value % 10 + '0');
107  value = T(value / 10);
108  } while (value);
109 
110  // and dump it in the right order
111  writeRaw(begin, end);
112  }

114  void writeDecimals(uint32_t value, int8_t width) {
115  // buffer should be big enough for all digits and the dot
116  char buffer[16];
117  char *end = buffer + sizeof(buffer);
118  char *begin = end;
119 
120  // write the string in reverse order
121  while (width--) {
122  *--begin = char(value % 10 + '0');
123  value /= 10;

127  // and dump it in the right order
128  writeRaw(begin, end);
