70 void spi_xfr(void *txdata, void *rxdata, size_t n) {
71  uint32_t temp_CHCTRLB_reg;
72 
73  // set up transmit channel
74  DMAC->CHID.reg = DMAC_CHID_ID(chnltx);
75  DMAC->CHCTRLA.reg &= ~DMAC_CHCTRLA_ENABLE;
76  DMAC->CHCTRLA.reg = DMAC_CHCTRLA_SWRST;
77  DMAC->SWTRIGCTRL.reg &= (uint32_t)(~(1 << chnltx));
78  temp_CHCTRLB_reg = DMAC_CHCTRLB_LVL(0) |
79  DMAC_CHCTRLB_TRIGSRC(SERCOM4_DMAC_ID_TX) | DMAC_CHCTRLB_TRIGACT_BEAT;
80  DMAC->CHCTRLB.reg = temp_CHCTRLB_reg;
81  DMAC->CHINTENSET.reg = DMAC_CHINTENSET_MASK ; // enable all 3 interrupts
82  descriptor.descaddr = 0;
83  descriptor.dstaddr = (uint32_t) &sercom->SPI.DATA.reg;
84  descriptor.btcnt = n;
85  descriptor.srcaddr = (uint32_t)txdata;
86  descriptor.btctrl = DMAC_BTCTRL_VALID;
87  if (xtype != DoRX) {
88  descriptor.srcaddr += n;
89  descriptor.btctrl |= DMAC_BTCTRL_SRCINC;
90  }
91  memcpy(&descriptor_section[chnltx],&descriptor, sizeof(dmacdescriptor));
92 
93  // rx channel enable interrupts
94  DMAC->CHID.reg = DMAC_CHID_ID(chnlrx);
95  DMAC->CHCTRLA.reg &= ~DMAC_CHCTRLA_ENABLE;
96  DMAC->CHCTRLA.reg = DMAC_CHCTRLA_SWRST;
97  DMAC->SWTRIGCTRL.reg &= (uint32_t)(~(1 << chnlrx));
98  temp_CHCTRLB_reg = DMAC_CHCTRLB_LVL(0) |
99  DMAC_CHCTRLB_TRIGSRC(SERCOM4_DMAC_ID_RX) | DMAC_CHCTRLB_TRIGACT_BEAT;
100  DMAC->CHCTRLB.reg = temp_CHCTRLB_reg;
101  DMAC->CHINTENSET.reg = DMAC_CHINTENSET_MASK ; // enable all 3 interrupts
102  dmadone = 0;
103  descriptor.descaddr = 0;
104  descriptor.srcaddr = (uint32_t) &sercom->SPI.DATA.reg;
105  descriptor.btcnt = n;
106  descriptor.dstaddr = (uint32_t)rxdata;
107  descriptor.btctrl = DMAC_BTCTRL_VALID;
108  if (xtype != DoTX) {
109  descriptor.dstaddr += n;
110  descriptor.btctrl |= DMAC_BTCTRL_DSTINC;
111  }
112  memcpy(&descriptor_section[chnlrx],&descriptor, sizeof(dmacdescriptor));
113 
114  // start both channels ? order matter ?
115  DMAC->CHID.reg = DMAC_CHID_ID(chnltx);
116  DMAC->CHCTRLA.reg |= DMAC_CHCTRLA_ENABLE;
117  DMAC->CHID.reg = DMAC_CHID_ID(chnlrx);
118  DMAC->CHCTRLA.reg |= DMAC_CHCTRLA_ENABLE;
119 
120  while(!dmadone); // await DMA done isr
121 
122  DMAC->CHID.reg = DMAC_CHID_ID(chnltx); //disable DMA to allow lib SPI
123  DMAC->CHCTRLA.reg &= ~DMAC_CHCTRLA_ENABLE;
124  DMAC->CHID.reg = DMAC_CHID_ID(chnlrx);
125  DMAC->CHCTRLA.reg &= ~DMAC_CHCTRLA_ENABLE;
126 }
